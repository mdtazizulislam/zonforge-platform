const leadForm = document.getElementById('lead-form');
const leadStatus = document.getElementById('lead-status');

function encodeFormData(formData) {
  return new URLSearchParams(Array.from(formData.entries())).toString();
}

if (leadForm) {
  leadForm.addEventListener('submit', async (event) => {
    event.preventDefault();

    const btn = leadForm.querySelector('.df-submit');
    const originalLabel = btn.textContent;
    const formData = new FormData(leadForm);
    formData.set('form-name', 'lead-capture');

    btn.disabled = true;
    btn.textContent = 'Submitting...';

    try {
      const response = await fetch('/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
        body: encodeFormData(formData),
      });

      if (!response.ok) {
        throw new Error('Lead capture request failed');
      }

      leadForm.reset();
      leadStatus.style.display = 'block';
      leadStatus.style.color = '#22c55e';
      leadStatus.textContent = '✓ Request received. Our team will contact you at support@zonforge.com within 2 hours.';
      btn.textContent = '✓ Request Received';
    } catch (error) {
      leadStatus.style.display = 'block';
      leadStatus.style.color = '#f59e0b';
      leadStatus.textContent = 'Submission could not be completed. Email support@zonforge.com and we will set you up manually.';
      btn.disabled = false;
      btn.textContent = originalLabel;
    }
  });
}

function switchTab(tab) {
